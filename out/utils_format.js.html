"use strict";

/**
 *  @module utils/Fmt
 */
module.exports = {
/*
 * @function diffObj
 *
 */
  diffObj: (rRev, rNext, lPrev, lNext, rList, lList) => {
    return { coords: {
            r: {prev:rRev, next:rNext},
            l: {prev:lPrev, next:lNext}
          },
          del: rList,
          add: lList
        }
  },

/*
 * @function segObj
 *
 */
  segObj: (type, coord, rList, lList) => {
    return { coords: {
            r: {prev:rRev, next:rNext},
            l: {prev:lPrev, next:lNext}
          },
          del: rList,
          add: lList
        }
  },

/*
 * @function segPhrase
 *
 */
  segPhrase: (lineNo, start, end, words) => {
    return {
          line: lineNo, 
          coord: {
            start: start,
            end: end
          },
          words: words
        }
  }
}
